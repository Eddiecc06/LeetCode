# 1339. Maximum Product of Splitted Binary Tree
[Medium](https://leetcode.com/problems/maximum-product-of-splitted-binary-tree/description/)

## Problem Description

Given the root of a binary tree, split the binary tree into two subtrees by removing one edge such that the product of the sums of the subtrees is maximized.

Return the maximum product of the sums of the two subtrees. Since the answer may be too large, return it modulo 109 + 7.

Note that you need to maximize the answer before taking the mod and not after taking it.

## Solution

### _Related Topic_
   Tree, Depth-First Search, Binary Tree


### _C++ Code_
```cpp
/**
 * Definition for a binary tree node.
 * struct TreeNode {
 *     int val;
 *     TreeNode *left;
 *     TreeNode *right;
 *     TreeNode() : val(0), left(nullptr), right(nullptr) {}
 *     TreeNode(int x) : val(x), left(nullptr), right(nullptr) {}
 *     TreeNode(int x, TreeNode *left, TreeNode *right) : val(x), left(left), right(right) {}
 * };
 */
class Solution {
private:
    int mod = 1e9 + 7;
public:
    int maxProduct(TreeNode* root) {
        long long totalSum = 0, ans = 0;
        GetTotalSum(root, totalSum);
        FindMaxProduct(root, totalSum, ans);
        return ans % mod;
    }
    void GetTotalSum(TreeNode* root, long long &totalSum){
        if(root == nullptr)
            return;
        totalSum = (totalSum + root->val) % mod;
        if(root->left)
            GetTotalSum(root->left, totalSum);
        if(root->right)
            GetTotalSum(root->right, totalSum);
    }
    long long FindMaxProduct(TreeNode* root, long long &totalSum, long long &maxProduct){
        if(root == nullptr)
            return 0;
        long long leftTreeSum = FindMaxProduct(root->left, totalSum, maxProduct);
        long long rightTreeSum = FindMaxProduct(root->right, totalSum, maxProduct);
        long long subTreeSum = leftTreeSum + rightTreeSum + root->val;
        long long remainTreeSum = totalSum - subTreeSum;
        maxProduct = max(maxProduct, (remainTreeSum * subTreeSum));
        return subTreeSum;
    }

};
```

### _Complexity Anlysis_
- _Time Complexity_: O(n)
- _Space Complexity_：O(n)
